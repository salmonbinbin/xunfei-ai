> 本文件是教师端开发指南，定义代码规范、API设计、服务设计规范。

## 一、概述

### 1.1 与学生端的关系

- 共用同一后端架构，通过 `role` 字段区分用户
- 教师端功能独立于学生端，但共享基础服务（如讯飞API、数据库）
- 前端独立路由 `/teacher/*`

### 1.2 技术栈

与学生端完全一致：
- 前端：Vue.js 3 + Element Plus + Pinia
- 后端：FastAPI + SQLAlchemy + aiomysql
- AI：讯飞开放平台

### 2.3 服务编写规范

```python
# backend/app/services/grade_service.py

from typing import List, Dict, Optional
from openpyxl import load_workbook
from io import BytesIO

class GradeService:
    """成绩管理服务"""

    def __init__(self):
        self.logger = logging.getLogger("grade")

    async def parse_excel(self, file_data: bytes) -> List[Dict]:
        """
        解析Excel成绩表

        Args:
            file_data: Excel文件二进制数据

        Returns:
            List[Dict]: 成绩列表
            示例：[{'student_name': '张三', 'usual_score': 85, ...}, ...]
        """
        self.logger.info(f"[Grade] Parsing excel, size: {len(file_data)} bytes")

        try:
            wb = load_workbook(BytesIO(file_data))
            ws = wb.active

            # 读取表头行
            headers = [cell.value for cell in ws[1]]

            # 映射列名
            # expected: ['学生姓名', '学号', '平时分', '期中分', '期末分', '实验分']
            results = []
            for row in ws.iter_rows(min_row=2, values_only=True):
                item = {}
                for i, header in enumerate(headers):
                    if header:
                        item[header] = row[i]
                results.append(item)

            self.logger.info(f"[Grade] Parsed {len(results)} items")
            return results

        except Exception as e:
            self.logger.error(f"[Grade] Parse failed: {e}", exc_info=True)
            raise ValidationException("Excel解析失败", details={"error": str(e)})

    async def calculate_grades(
        self,
        items: List[Dict],
        weights: Optional[Dict[str, float]] = None
    ) -> List[Dict]:
        """
        计算总评成绩

        Args:
            items: 原始成绩列表
            weights: 权重配置，默认 {usual: 0.4, midterm: 0.2, final: 0.4}

        Returns:
            含总评和排名的成绩列表
        """
        if weights is None:
            weights = {"usual": 0.4, "midterm": 0.2, "final": 0.4, "practice": 0.0}

        self.logger.info(f"[Grade] Calculating {len(items)} grades with weights: {weights}")

        # 计算总评
        for item in items:
            total = 0.0
            weight_sum = 0.0

            for key, weight in weights.items():
                score = item.get(f"{key}_score")
                if score is not None and weight > 0:
                    total += float(score) * weight
                    weight_sum += weight

            item["total_score"] = round(total / weight_sum if weight_sum > 0 else 0, 2)

        # 计算排名
        sorted_items = sorted(items, key=lambda x: x.get("total_score", 0), reverse=True)
        for rank, item in enumerate(sorted_items, 1):
            item["rank"] = rank

        return items

    async def calculate_stats(self, items: List[Dict]) -> Dict:
        """
        计算统计数据

        Returns:
            {
                distribution: [{range, count}, ...],  # 成绩分布
                avg_score: float,                       # 平均分
                pass_rate: float,                      # 及格率
                max_score: float,
                min_score: float
            }
        """
        if not items:
            return {"distribution": [], "avg_score": 0, "pass_rate": 0}

        total_scores = [item.get("total_score", 0) for item in items]

        # 成绩分布
        ranges = ["<60", "60-70", "70-80", "80-90", ">90"]
        distribution = []
        for r in ranges:
            if r == "<60":
                count = sum(1 for s in total_scores if s < 60)
            elif r == ">90":
                count = sum(1 for s in total_scores if s >= 90)
            else:
                low, high = map(int, r.split("-"))
                count = sum(1 for s in total_scores if low <= s < high)
            distribution.append({"range": r, "count": count})

        return {
            "distribution": distribution,
            "avg_score": round(sum(total_scores) / len(total_scores), 2),
            "pass_rate": round(sum(1 for s in total_scores if s >= 60) / len(total_scores), 2),
            "max_score": max(total_scores),
            "min_score": min(total_scores)
        }
```

```python
# backend/app/services/notification_service.py

from app.services.xinghuo_service import xinghuo_service

class NotificationService:
    """通知生成服务"""

    async def generate(self, topic: str, context: str = "") -> Dict:
        """
        AI生成通知

        Args:
            topic: 通知主题/关键词
            context: 补充上下文（如考试科目、时间等）

        Returns:
            {title, content, deadline_hint}
        """
        self.logger.info(f"[Notification] Generating for topic: {topic}")

        prompt = f"""你是一个专业的校园通知撰写助手。请根据以下主题生成一份格式规范的通知。

主题：{topic}
{context}

要求：
1. 标题简洁明了，包含【通知】前缀
2. 正文语气正式，格式规范（使用序号分点）
3. 包含适当的落款（广州商学院XX学院 + 日期）
4. 如果涉及截止时间，给出合理的建议

请返回JSON格式：
{{"title": "通知标题", "content": "通知正文", "deadline_hint": "建议的截止时间"}}
"""

        result = await xinghuo_service.chat(prompt)
        # 解析result为JSON

        return {
            "title": "关于xxx的通知",  # 解析自result
            "content": "...",           # 解析自result
            "deadline_hint": "2026年5月20日"
        }
```

## 八、注意事项

1. **角色隔离**：所有教师端API需在路由层验证 `role=teacher`
2. **文件大小限制**：Excel上传限制 5MB
3. **PPT页数限制**：单次生成不超过 50 页
4. **成绩权重**：默认 40%平时 + 20%期中 + 40%期末
5. **数据隔离**：教师只能查看自己上传的成绩
