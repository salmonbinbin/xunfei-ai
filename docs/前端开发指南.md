## 二、核心组件

### 2.1 VoiceInput.vue（语音输入）

```vue
<template>
  <div class="voice-input">
    <button
      class="mic-button"
      :class="{ recording: isRecording }"
      @mousedown="startRecording"
      @mouseup="stopRecording"
      @touchstart="startRecording"
      @touchend="stopRecording"
    >
      <span class="mic-icon">🎤</span>
    </button>
    <div v-if="isRecording" class="waveform">
      <span v-for="i in 5" :key="i" class="wave" :style="{ animationDelay: `${i * 0.1}s` }"></span>
    </div>
    <div v-if="recognizedText" class="recognized-text">
      {{ recognizedText }}
    </div>
  </div>
</template>

<script setup>
import { ref } from 'vue'

const props = defineProps({
  // 是否在录音
})

const emit = defineEmits(['result', 'error'])

const isRecording = ref(false)
const recognizedText = ref('')
let mediaRecorder = null
let audioChunks = []

async function startRecording() {
  try {
    const stream = await navigator.mediaDevices.getUserMedia({ audio: true })
    mediaRecorder = new MediaRecorder(stream)
    audioChunks = []

    mediaRecorder.ondataavailable = (e) => {
      audioChunks.push(e.data)
    }

    mediaRecorder.onstop = async () => {
      const audioBlob = new Blob(audioChunks, { type: 'audio/webm' })
      await uploadAudio(audioBlob)
      stream.getTracks().forEach(track => track.stop())
    }

    mediaRecorder.start()
    isRecording.value = true
    recognizedText.value = ''
  } catch (error) {
    emit('error', error)
  }
}

function stopRecording() {
  if (mediaRecorder && isRecording.value) {
    mediaRecorder.stop()
    isRecording.value = false
  }
}

async function uploadAudio(blob) {
  const formData = new FormData()
  formData.append('audio', blob, 'recording.webm')
  // 调用后端 API
  const response = await api.post('/chat/voice', formData)
  recognizedText.value = response.data.recognized_text
  emit('result', {
    text: response.data.recognized_text,
    voiceUrl: response.data.voice_url
  })
}
</script>
```

### 2.2 AIBubble.vue（对话气泡）

```vue
<template>
  <div class="message-bubble" :class="[role, { streaming }]">
    <div class="avatar">{{ role === 'assistant' ? '🤖' : '👤' }}</div>
    <div class="content">
      <div class="text">{{ displayText }}</div>
      <div v-if="role === 'assistant' && audioUrl" class="audio-player">
        <button @click="playAudio">🔊 播放</button>
      </div>
      <div v-if="sentiment && sentiment !== 'neutral'" class="sentiment-badge">
        {{ sentimentLabel }}
      </div>
    </div>
  </div>
</template>

<script setup>
import { ref, computed, watch } from 'vue'

const props = defineProps({
  role: { type: String, default: 'assistant' }, // user / assistant
  content: { type: String, default: '' },
  audioUrl: { type: String, default: null },
  sentiment: { type: String, default: null },
  streaming: { type: Boolean, default: false }
})

const displayText = ref('')
watch(() => props.content, (newVal) => {
  if (props.streaming) {
    // 流式：逐步显示
    displayText.value = newVal
  } else {
    displayText.value = newVal
  }
}, { immediate: true })

const sentimentLabel = computed(() => ({
  positive: '😊 积极',
  negative: '😔 消极',
  neutral: '😐 中性'
}[props.sentiment] || ''))

function playAudio() {
  if (props.audioUrl) {
    new Audio(props.audioUrl).play()
  }
}
</script>
```

## 三、API 封装示例

```javascript
// frontend/src/api/chat.js

import api from './index'

/**
 * 发送文字消息
 * @param {Object} params
 * @param {number} params.conv_id - 会话ID
 * @param {string} params.message - 消息内容
 * @param {string} [params.mode] - normal/emotion
 * @param {boolean} [params.stream] - 是否流式
 */
export function sendMessage(params) {
  return api.post('/chat/message', params)
}

/**
 * 发送语音消息
 * @param {FormData} formData
 */
export function sendVoice(formData) {
  return api.post('/chat/voice', formData, {
    headers: { 'Content-Type': 'multipart/form-data' }
  })
}

/**
 * 获取对话历史
 * @param {number} convId
 */
export function getConversation(convId) {
  return api.get(`/chat/conversation/${convId}`)
}
```

## 四、状态管理

```javascript
// frontend/src/stores/chat.js

import { defineStore } from 'pinia'
import { ref } from 'vue'
import { sendMessage, sendVoice } from '@/api/chat'

export const useChatStore = defineStore('chat', () => {
  // 状态
  const messages = ref([])
  const currentConvId = ref(null)
  const isLoading = ref(false)
  const mode = ref('normal') // normal / emotion

  // Actions
  async function send(content, audioBlob = null) {
    isLoading.value = true

    try {
      let result
      if (audioBlob) {
        const formData = new FormData()
        formData.append('audio', audioBlob, 'recording.webm')
        formData.append('mode', mode.value)
        result = await sendVoice(formData)
      } else {
        result = await sendMessage({
          conv_id: currentConvId.value,
          message: content,
          mode: mode.value,
          stream: false
        })
      }

      // 添加用户消息
      messages.value.push({ role: 'user', content })

      // 添加AI回复
      messages.value.push({
        role: 'assistant',
        content: result.data.reply,
        sentiment: result.data.sentiment
      })

      return result
    } finally {
      isLoading.value = false
    }
  }

  function toggleMode() {
    mode.value = mode.value === 'normal' ? 'emotion' : 'normal'
  }

  function clearMessages() {
    messages.value = []
  }

  return {
    messages,
    currentConvId,
    isLoading,
    mode,
    send,
    toggleMode,
    clearMessages
  }
})
```

### 5.1 useAsync 组合式函数

```javascript
// frontend/src/utils/useAsync.js

import { ref } from 'vue'
import { ElMessage } from 'element-plus'
import { handleError } from './errorHandler'

/**
 * 异步操作封装
 */
export function useAsync(asyncFn, options = {}) {
  const { immediate = false } = options

  const data = ref(null)
  const loading = ref(false)
  const error = ref(null)

  async function execute(...args) {
    loading.value = true
    error.value = null

    try {
      const result = await asyncFn(...args)
      data.value = result
      return result
    } catch (e) {
      error.value = e
      handleError(e)
      throw e
    } finally {
      loading.value = false
    }
  }

  if (immediate) {
    execute()
  }

  return { data, loading, error, execute }
}
```

## 六、路由配置

```javascript
// frontend/src/router/index.js

import { createRouter, createWebHistory } from 'vue-router'
import { useUserStore } from '@/stores/user'

const routes = [
  { path: '/login', name: 'Login', component: () => import('@/views/Login.vue') },
  { path: '/guide', name: 'Guide', component: () => import('@/views/Guide.vue') },
  {
    path: '/',
    component: () => import('@/layouts/MainLayout.vue'),
    meta: { requiresAuth: true },
    children: [
      { path: '', name: 'Home', component: () => import('@/views/Home.vue') },
      { path: 'timetable', name: 'Timetable', component: () => import('@/views/Timetable.vue') },
      { path: 'timetable/import', name: 'TimetableImport', component: () => import('@/views/TimetableImport.vue') },
      { path: 'schedule', name: 'Schedule', component: () => import('@/views/Schedule.vue') },
      { path: 'review', name: 'Review', component: () => import('@/views/Review.vue') },
      { path: 'review/:id', name: 'ReviewDetail', component: () => import('@/views/ReviewDetail.vue') },
      { path: 'ai-sister', name: 'AISister', component: () => import('@/views/AISister.vue') },
    ]
  },
  {
    path: '/admin',
    component: () => import('@/layouts/AdminLayout.vue'),
    meta: { requiresAuth: true, requiresAdmin: true },
    children: [
      { path: '', name: 'Dashboard', component: () => import('@/views/admin/Dashboard.vue') },
      { path: 'knowledge', name: 'Knowledge', component: () => import('@/views/admin/Knowledge.vue') },
      { path: 'users', name: 'Users', component: () => import('@/views/admin/Users.vue') },
    ]
  }
]

const router = createRouter({
  history: createWebHistory(),
  routes
})

// 路由守卫
router.beforeEach((to, from, next) => {
  const userStore = useUserStore()

  if (to.meta.requiresAuth && !userStore.isLoggedIn) {
    next('/login')
  } else if (to.meta.requiresAdmin && !userStore.isAdmin) {
    next('/')
  } else {
    next()
  }
})

export default router
```

## 七、开发规范

1. **组件命名**：使用 PascalCase，如 `VoiceInput.vue`
2. **API封装**：一个功能模块一个文件，如 `chat.js`、`timetable.js`
3. **状态管理**：使用 Pinia，状态按模块划分
4. **样式**：使用 SCSS，统一使用 Element Plus 主题变量
5. **错误处理**：所有 API 调用必须使用 `useAsync` 或 try-catch
