### 2.2 用户操作日志（user_logs）

**记录时机：** 用户每次使用功能时

**表结构：**
```sql
CREATE TABLE user_logs (
    id INT PRIMARY KEY AUTO_INCREMENT,
    user_id INT NOT NULL,
    user_type ENUM('student', 'teacher') NOT NULL,
    action VARCHAR(100) NOT NULL COMMENT '操作类型：ai_chat, voice_input, upload_audio, etc',
    module VARCHAR(50) NOT NULL COMMENT '模块：ai_sister, review, timetable, translate, activity',
    duration_ms INT DEFAULT 0 COMMENT '操作耗时（毫秒）',
    success TINYINT DEFAULT 1 COMMENT '是否成功',
    error_msg TEXT COMMENT '错误信息',
    ip_address VARCHAR(50) COMMENT '用户IP',
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,

    INDEX idx_user_id (user_id),
    INDEX idx_action (action),
    INDEX idx_module (module),
    INDEX idx_created_at (created_at)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;
```

**action 枚举值：**
| action | 说明 |
|--------|------|

| translate_doc | 文档翻译 |

**module 枚举值：**

## 三、数据统计方案

### 3.1 统计SQL

**今日活跃用户数（去重）：**
```sql
SELECT COUNT(DISTINCT user_id)
FROM user_logs
WHERE DATE(created_at) = CURDATE()
```

**今日在线人数（去重登录用户）：**
```sql
SELECT COUNT(DISTINCT user_id)
FROM login_logs
WHERE DATE(created_at) = CURDATE()
```

**今日登录学生数：**
```sql
SELECT COUNT(DISTINCT user_id)
FROM login_logs
WHERE DATE(created_at) = CURDATE()
  AND user_type = 'student'
```

**今日登录教师数：**
```sql
SELECT COUNT(DISTINCT user_id)
FROM login_logs
WHERE DATE(created_at) = CURDATE()
  AND user_type = 'teacher'
```

**功能使用排行（TOP5）：**
```sql
SELECT module, COUNT(*) as count
FROM user_logs
WHERE DATE(created_at) = CURDATE()
GROUP BY module
ORDER BY count DESC
LIMIT 5
```

**API调用统计：**
```sql
SELECT
    api_name,
    COUNT(*) as total_count,
    SUM(CASE WHEN call_type = 'fail' THEN 1 ELSE 0 END) as fail_count,
    ROUND(SUM(CASE WHEN call_type = 'fail' THEN 1 ELSE 0 END) / COUNT(*) * 100, 2) as fail_rate,
    AVG(response_time_ms) as avg_response_time
FROM api_logs
WHERE DATE(created_at) = CURDATE()
GROUP BY api_name
```

**24小时活跃趋势：**
```sql
SELECT
    HOUR(created_at) as hour,
    COUNT(DISTINCT user_id) as user_count
FROM user_logs
WHERE created_at >= CURDATE()
  AND created_at < CURDATE() + INTERVAL 1 DAY
GROUP BY HOUR(created_at)
ORDER BY hour
```

### 3.2 统计服务实现

```python
# backend/app/services/stats_service.py

class StatsService:
    """统计数据服务"""

    async def get_dashboard_stats(self, date: date = None) -> Dict[str, Any]:
        """获取仪表盘统计数据"""
        if date is None:
            date = datetime.now().date()

        start_of_day = datetime.combine(date, datetime.min.time())
        end_of_day = datetime.combine(date, datetime.max.time())

        # 并行查询各项统计
        tasks = [
            self._count_active_users(start_of_day, end_of_day),
            self._count_online_users(start_of_day, end_of_day),
            self._count_student_logins(start_of_day, end_of_day),
            self._count_teacher_logins(start_of_day, end_of_day),
            self._get_api_stats(start_of_day, end_of_day),
            self._get_module_usage_ranking(start_of_day, end_of_day),
            self._get_hourly_trend(start_of_day, end_of_day),
        ]

        results = await asyncio.gather(*tasks)

        return {
            "today_active_users": results[0],
            "today_online_users": results[1],
            "today_student_logins": results[2],
            "today_teacher_logins": results[3],
            "api_stats": results[4],
            "module_ranking": results[5],
            "hourly_trend": results[6],
            "date": date.isoformat()
        }

    async def _count_active_users(self, start: datetime, end: datetime) -> int:
        """活跃用户数（去重）"""
        query = select(func.count(func.distinct(user_logs.c.user_id))).where(
            user_logs.c.created_at >= start,
            user_logs.c.created_at <= end
        )
        result = await db.execute(query)
        return result.scalar() or 0

    async def _count_online_users(self, start: datetime, end: datetime) -> int:
        """在线人数（去重登录用户）"""
        query = select(func.count(func.distinct(login_logs.c.user_id))).where(
            login_logs.c.created_at >= start,
            login_logs.c.created_at <= end
        )
        result = await db.execute(query)
        return result.scalar() or 0
```

## 六、API调用埋点实现

### 6.1 统一拦截器

在调用讯飞API的地方统一记录日志：

```python
# backend/app/services/xinghuo_service.py

class XingHuoService:
    """讯飞服务 - 包含API调用日志记录"""

    async def chat_completion(self, messages: List[Dict], user_id: int = None) -> str:
        """星火大模型对话"""
        start_time = time.time()

        try:
            result = await self._do_chat(messages)
            response_time = int((time.time() - start_time) * 1000)

            # 记录成功日志
            await self._log_api_call(
                api_name="spark",
                call_type="success",
                response_time_ms=response_time,
                user_id=user_id
            )

            return result

        except Exception as e:
            response_time = int((time.time() - start_time) * 1000)

            # 记录失败日志
            await self._log_api_call(
                api_name="spark",
                call_type="fail",
                response_time_ms=response_time,
                error_msg=str(e),
                user_id=user_id
            )
            raise

    async def _log_api_call(self, **kwargs):
        """记录API调用日志"""
        from app.services.api_log_service import save_api_log
        await save_api_log(**kwargs)
```

## 七、图表配置

### 7.1 ECharts 组件封装

```javascript
// frontend/src/utils/echarts.js

import * as echarts from 'echarts'

// 主题色
const colors = {
  primary: '#0891B2',
  success: '#059669',
  warning: '#F59E0B',
  danger: '#EF4444',
  info: '#8B5CF6'
}

// 柱状图配置
export function getBarChartOption(data) {
  return {
    tooltip: { trigger: 'axis' },
    grid: { left: '3%', right: '4%', bottom: '3%', containLabel: true },
    xAxis: {
      type: 'category',
      data: data.map(d => d.name),
      axisLabel: { color: '#475569' }
    },
    yAxis: {
      type: 'value',
      axisLabel: { color: '#475569' }
    },
    series: [{
      type: 'bar',
      data: data.map(d => d.value),
      itemStyle: {
        color: new echarts.graphic.LinearGradient(0, 0, 0, 1, [
          { offset: 0, color: colors.primary },
          { offset: 1, color: '#22D3EE' }
        ]),
        borderRadius: [4, 4, 0, 0]
      }
    }]
  }
}

// 饼图配置
export function getPieChartOption(data) {
  return {
    tooltip: { trigger: 'item', formatter: '{b}: {c} ({d}%)' },
    legend: {
      orient: 'vertical',
      right: '5%',
      top: 'center',
      textStyle: { color: '#475569' }
    },
    series: [{
      type: 'pie',
      radius: ['40%', '70%'],
      center: ['35%', '50%'],
      avoidLabelOverlap: false,
      label: { show: false },
      emphasis: {
        label: { show: true, fontSize: 16, fontWeight: 'bold' }
      },
      data: data.map((d, i) => ({
        value: d.value,
        name: d.name,
        itemStyle: { color: Object.values(colors)[i % Object.keys(colors).length] }
      }))
    }]
  }
}

// 折线图配置
export function getLineChartOption(data) {
  return {
    tooltip: { trigger: 'axis' },
    grid: { left: '3%', right: '4%', bottom: '3%', containLabel: true },
    xAxis: {
      type: 'category',
      data: data.map(d => d.hour + ':00'),
      boundaryGap: false,
      axisLabel: { color: '#475569' }
    },
    yAxis: {
      type: 'value',
      axisLabel: { color: '#475569' }
    },
    series: [{
      type: 'line',
      data: data.map(d => d.count),
      smooth: true,
      areaStyle: {
        color: new echarts.graphic.LinearGradient(0, 0, 0, 1, [
          { offset: 0, color: 'rgba(8, 145, 178, 0.3)' },
          { offset: 1, color: 'rgba(8, 145, 178, 0.05)' }
        ])
      },
      lineStyle: { color: colors.primary, width: 2 },
      itemStyle: { color: colors.primary }
    }]
  }
}
```
