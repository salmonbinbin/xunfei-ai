### 2.4 OCR文字识别服务

```python
# backend/app/services/ocr_service.py

import base64
import json
import logging
from app.config import settings
from app.utils.errors import ThirdPartyException

logger = logging.getLogger("xfyun")

class OCRService:
    """讯飞文字识别服务"""

    async def recognize_general(self, image_path: str) -> str:
        """
        通用文字识别
        返回：图片中的所有文字（按行组织）
        """
        url = "https://webapi.xfyun.cn/v1/service/v1/ocr/general"

        with open(image_path, "rb") as f:
            image_data = f.read()

        body = {
            "image": base64.b64encode(image_data).decode('utf-8'),
            "image_type": "base64",
            "param": base64.b64encode(json.dumps({
                "language": "cn|en",
                "location": "false"
            }).encode('utf-8')).decode('utf-8')
        }

        headers = self._build_auth_headers(url)
        response = requests.post(url, headers=headers, data=body, timeout=15)
        result = response.json()

        if result.get("code") != 0:
            raise ThirdPartyException("OCR", result.get("desc"))

        # 提取文字行
        lines = []
        for block in result.get("data", {}).get("block", []):
            for line in block.get("line", []):
                line_text = "".join([w.get("content", "") for w in line.get("word", [])])
                lines.append(line_text)

        return "\n".join(lines)
```

### 2.5 NLP情感分析服务

```python
# backend/app/services/nlp_service.py

import base64
import logging
from app.config import settings

logger = logging.getLogger("xfyun")

class NLPService:
    """讯飞NLP能力服务"""

    async def sentiment_analysis(self, text: str) -> dict:
        """
        情感分析
        返回：{
            "sentiment": "positive/neutral/negative",
            "confidence": 0.95,
            "suggest_emotion_mode": True/False
        }
        """
        url = "https://nlp-api.xfyun.cn/v1/sa"

        body = {
            "type": "dependent",
            "text": base64.b64encode(text.encode('utf-8')).decode('utf-8'),
        }

        headers = self._build_auth_headers(url)
        response = requests.post(url, headers=headers, json=body, timeout=10)
        result = response.json()

        if result.get("code") != 0:
            logger.warning(f"[NLP] sentiment analysis failed: {result}")
            return {"sentiment": "neutral", "confidence": 0.5, "suggest_emotion_mode": False}

        sentiment_data = result.get("data", {}).get("sentiment_list", [{}])[0]
        sentiment = sentiment_data.get("sentiment", 1)  # 0=negative, 1=neutral, 2=positive

        sentiment_map = {0: "negative", 1: "neutral", 2: "positive"}
        sentiment_str = sentiment_map.get(sentiment, "neutral")

        return {
            "sentiment": sentiment_str,
            "confidence": sentiment_data.get("confidence", 0.5),
            "suggest_emotion_mode": (sentiment == 0)  # 负面情绪建议切换情感模式
        }
```

### 3.1 基础模型

```python
# backend/app/models/base.py

from sqlalchemy import Column, Integer, DateTime, Boolean
from sqlalchemy.ext.declarative import declarative_base
from datetime import datetime

Base = declarative_base()

class TimestampMixin:
    """时间戳混入"""
    created_at = Column(DateTime, default=datetime.utcnow)
    updated_at = Column(DateTime, default=datetime.utcnow, onupdate=datetime.utcnow)
    is_deleted = Column(Boolean, default=False)

class BaseModel(Base, TimestampMixin):
    """所有模型的基类"""
    __abstract__ = True

    id = Column(Integer, primary_key=True, autoincrement=True)

    @classmethod
    def get_by_id(cls, db, id):
        return db.query(cls).filter(cls.id == id, cls.is_deleted == False).first()
```

from fastapi import APIRouter, Depends, Request
from fastapi.responses import StreamingResponse
from sqlalchemy.orm import Session

from app.schemas.chat import ChatRequest, ChatResponse

from app.services.nlp_service import nlp_service
from app.services.knowledge_base import knowledge_base
from app.utils.errors import handle_app_errors
from app.utils.logging import get_logger

router = APIRouter(prefix="/api/chat", tags=["AI对话"])
logger = get_logger("api")

@router.post("/message")
@handle_app_errors
async def chat_message(
    request: ChatRequest,
    current_user: User = Depends(get_current_user),
    db: Session = Depends(get_db)
):
    """发送消息，获取AI学姐回复"""
    # 1. 情感分析
    nlp_result = await nlp_service.sentiment_analysis(request.message)

    # 2. RAG知识库检索
    retrieved_knowledge = await knowledge_base.search(request.message, top_k=3)

    # 3. 构建提示词
    system_prompt = _build_system_prompt(current_user, request.mode)
    messages = [{"role": "system", "content": system_prompt}]
    messages.extend([
        {"role": m.role, "content": m.content}
        for m in get_recent_messages(db, request.conv_id, limit=10)
    ])
    messages.append({"role": "user", "content": request.message})

    # 4. 调用星火
    reply = await xinghuo_service.chat_completion(messages)

    return {
        "success": True,
        "data": {
            "reply": reply,
            "conv_id": request.conv_id or 1,
            "sentiment": nlp_result["sentiment"]
        }
    }

先共情后引导，用温暖的语言，不说教、不讲大道理。"""

    profile = user.profile
    return f"""你是"AI小商学姐"，广州商学院的智能校园助手。
专业：{profile.major if profile else '未知'}
年级：大{profile.grade if profile and profile.grade else 'X'}年级
目标：{profile.goal if profile else '未设定'}"""

## 五、开发规范

1. **配置管理**：所有配置从 `app/config.py` 导入，不硬编码
2. **错误处理**：所有路由函数使用 `@handle_app_errors` 装饰器
3. **日志记录**：使用 `get_logger()` 获取专用Logger
4. **讯飞调用**：必须实现重试机制
5. **数据库**：使用 SQLAlchemy ORM，不直接写SQL
6. **模型定义**：统一使用 `BaseModel` 基类
